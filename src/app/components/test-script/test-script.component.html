package app.components.test-script; import { Component } from '@angular/core'; @Component({ selector: 'app-test-script', templateUrl: './test-script.component.html', }) export class TestScriptComponent { numTerms: number; fibonacciResult: number[] = []; errorMessage: string = ''; calculateFibonacci() { this.errorMessage = ''; if (!this.isValidInput()) { this.errorMessage = 'Please enter a valid number greater than 0.'; return; } this.fibonacciResult = this.generateFibonacci(this.numTerms); } isValidInput(): boolean { return this.numTerms > 0; } generateFibonacci(n: number): number[] { const result = []; let a = 0, b = 1; for (let i = 0; i < n; i++) { result.push(a); [a, b] = [b, a + b]; } return result; } }
