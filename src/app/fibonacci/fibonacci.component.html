package fibonacci; import { Component } from '@angular/core'; import { HttpClient } from '@angular/common/http'; import { NgForm } from '@angular/forms'; @Component({ selector: 'app-fibonacci', templateUrl: './fibonacci.component.html', styleUrls: ['./fibonacci.component.css'] }) export class FibonacciComponent { number: number; responseMessage: string; constructor(private http: HttpClient) {} onSubmit(form: NgForm) { if (form.valid) { this.http.post('https://api.example.com/fibonacci', { number: this.number }) .subscribe(response => { this.responseMessage = response['message'] || 'No message returned'; }); } } }
